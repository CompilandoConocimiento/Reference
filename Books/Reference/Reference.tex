                \begin{lstlisting}[language=C++, gobble=20]
                    #include <iomanip>
                    #include <iostream>

                    auto main() -> int {
                        using namespace std:

                        auto f = 301.14159;
                        cout << std::fixed;

                        cout << setprecision(5) << f << '\n';   // 301.14159
                        cout << setprecision(8) << f << '\n';   //  301.14159000     

                        return 0;
                    }
                \end{lstlisting}
